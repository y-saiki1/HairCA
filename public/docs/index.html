    Auth

    Login

    ユーザーログイン

    ログインする場合は以下のパラメータを用意すること

    Example request:

    curl -X POST "http://localhost/api/auth/login" \
    -H "Authorization: Bearer {token}" \
    -d "email"="example@exam.com" \
    -d "password"="password" 
    const url = new URL("http://localhost/api/auth/login");

let headers = {
    "Authorization": "Bearer {token}",
    "Accept": "application/json",
    "Content-Type": "application/json",
}

let body = JSON.stringify({
    "email": "example@exam.com",
    "password": "password",
})

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
})
    .then(response => response.json())
    .then(json => console.log(json));

    Example response (200):

    {
    "access_token": "aaaaaaaa",
    "token_type": "bearer",
    "expires_in": 3600
}

    Example response (400):

    {
    "message": "UnAuthorized"
}

    HTTP Request

    POST api/auth/login

    Body Parameters

    Parameter Type Status Description
    email string required ログインするアカウントのメールアドレス
    password string required ログインするアカウントのパスワード

    -H "Authorization: Bearer {token}" \
    -d "email"="example@exam.com" \
    -d "invitation_token"="token" 
    const url = new URL("http://localhost/api/accounts/stylists/auth");

let headers = {
    "Authorization": "Bearer {token}",
    "Accept": "application/json",
    "Content-Type": "application/json",
}

let body = JSON.stringify({
    "email": "example@exam.com",
    "invitation_token": "token",
})

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
})
    .then(response => response.json())
    .then(json => console.log(json));

    Example response (200):

    {
    "message": "The Guest that have this Email and Token is Stylist",
    "is_guest": true,
    "is_stylist": false,
    "is_member": false
}
